(function(exports) {
    "use strict";

    var Models = {};

    Models.User = function(first_name, last_name) {
        this.first_name = first_name
        this.last_name = last_name
    };

    Models.User.prototype.toString = function() {
        return this.first_name + ' ' + this.last_name;
    };

    exports.Models = Models;
})(this);

var makeUser = function(firstName, lastName) {
    return {
        firstName: firstName,
        lastName: lastName,
        toString: function() {
            return this.firstName + ' ' + this.lastName;
        }
    };
};

var tata = makeUser('Tata', 'de La Rochette de Rochegonde');
var toto = makeUser('Toto', "d'Aumont de Rochebaron");
            

var User = function(firstName, lastName) {
    this.firstName = firstName;
    this.lastName = lastName;
    this.toString = function() {
        return this.firstName + ' ' + this.lastName;

};
